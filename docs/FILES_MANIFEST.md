## 🚀 خطوات التثبيت السريعة

```bash
# 1. نسخ الملفات
mkdir -p src/components src/hooks docs

# Store
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/wifiNetwork.ts src/store/gameStore.ts

# Lib
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/wifiNetwork.ts src/lib/wifiNetwork.ts
cp /mnt/agents/output/soundEngine.ts src/lib/soundEngine.ts

# Screens
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/WifiGameScreen.tsx src/screens/WifiGameScreen.tsx
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/MainMenu.tsx src/screens/MainMenu.tsx

# Components
cp /mnt/agents/output/ErrorBoundary.tsx src/components/ErrorBoundary.tsx
cp /mnt/agents/output/ScreenTransition.tsx src/components/ScreenTransition.tsx
cp /mnt/agents/output/ExitConfirmation.tsx src/components/ExitConfirmation.tsx

# Hooks
cp /mnt/agents/output/useAndroidBackButton.ts src/hooks/useAndroidBackButton.ts

# App
cp /mnt/agents/output/App_Complete.tsx src/App.tsx

# Avatars
cp /mnt/agents/output/avatars/*.png public/assets/

# Docs
mkdir -p docs
cp /mnt/agents/output/sounds/SOUND_REQUIREMENTS.md docs/
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/WIFI_IMPLEMENTATION.md docs/
cp /mnt/agents/output/capacitor-plugin-wifi-p2p/INSTALLATION_GUIDE.md docs/
cp /mnt/agents/output/COMPLETE_FIXES_SUMMARY.md docs/

# 2. تثبيت الحزم
npm install @capacitor/preferences @capacitor/app zustand/middleware

# 3. مزامنة Capacitor
npx cap sync android

# 4. بناء
npm run build
cd android && ./gradlew assembleDebug
```
